USE master

IF EXISTS(SELECT * from sys.databases WHERE name='Location')
BEGIN
    DROP DATABASE Location;
END

CREATE DATABASE Location;
GO

USE Dblibrary;

CREATE TABLE dbo.Shelf
   (ID int IDENTITY(1,1) PRIMARY KEY NOT NULL,
    索取号 varchar(50) NOT NULL,
    条形码 varchar(50) NOT NULL)

CREATE TABLE dbo.BookOnShelf
   (ID int IDENTITY(1,1) PRIMARY KEY NOT NULL,
    正题名 varchar(25) NOT NULL,
    索取号 varchar(25) NOT NULL)

CREATE TABLE dbo.Map
   (ID int IDENTITY(1,1) PRIMARY KEY NOT NULL,
    floor varchar(25) NOT NULL,
	selection varchar(25) NOT NULL,
	row varchar(25) NOT NULL,
	col varchar(25) NOT NULL,
	col varchar(25) NOT NULL,
	col varchar(25) NOT NULL,
    索取号 varchar(25) NOT NULL)
